<script language="Javascript">
  function fibonacci(n){
    if (n <= 1)  {
      return n;
    } else {
      return fibonacci(n-1) + fibonacci(n-2);
    }
  }
  var n = 10;
  for (var i = 1; i <= n; i++) {
    document.writeln(i , ": ", fibonacci(i),"<br>");
  }
   
 
</script>
